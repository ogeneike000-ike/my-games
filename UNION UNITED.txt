<script>
// ---------- NAV / SECTIONS ----------
const navLinks = document.querySelectorAll("nav a[data-target]");
const sections = document.querySelectorAll(".section");
const gameCards = document.querySelectorAll(".game-card[data-target]");

function showSection(id){
sections.forEach(sec=>sec.classList.toggle("active",sec.id===id));
navLinks.forEach(link=>link.classList.toggle("active",link.getAttribute("data-target")===id));
}
navLinks.forEach(link=>{
link.addEventListener("click",()=>showSection(link.getAttribute("data-target")));
});
gameCards.forEach(card=>{
card.addEventListener("click",()=>showSection(card.getAttribute("data-target")));
});

// ---------- DATA & STORAGE ----------
const ADMIN_USERNAME = "IkeUnited";
const ADMIN_PASSWORD = "idwinbcimchrolo"; // replace locally later if you want

let users = JSON.parse(localStorage.getItem("users")) || [];
let stats = JSON.parse(localStorage.getItem("stats")) || {visits:0,plays:0,clicks:0};
let games = JSON.parse(localStorage.getItem("games")) || ["Neon Clicker"];
let logs = JSON.parse(localStorage.getItem("logs")) || [];
let maintenance = JSON.parse(localStorage.getItem("maintenance")) || false;
let globalMessage = localStorage.getItem("globalMessage") || "";
let currentUser = null;

function saveAll(){
localStorage.setItem("users",JSON.stringify(users));
localStorage.setItem("stats",JSON.stringify(stats));
localStorage.setItem("games",JSON.stringify(games));
localStorage.setItem("logs",JSON.stringify(logs));
localStorage.setItem("maintenance",JSON.stringify(maintenance));
localStorage.setItem("globalMessage",globalMessage);
}

// ensure admin exists
function ensureAdmin(){
const existing = users.find(u=>u.username===ADMIN_USERNAME);
if(!existing){
users.push({username:ADMIN_USERNAME,password:ADMIN_PASSWORD,role:"admin",banned:false});
addLog("System: created default admin account.");
saveAll();
}else{
existing.role="admin";
existing.banned=false;
saveAll();
}
}
ensureAdmin();

// ---------- LOGGING ----------
function addLog(text){
const ts = new Date().toLocaleTimeString();
logs.push("["+ts+"] "+text);
if(logs.length>200) logs = logs.slice(-200);
saveAll();
renderLogs();
}
function renderLogs(){
const logBox = document.getElementById("logBox");
if(!logBox) return;
logBox.innerHTML = logs.slice(-80).join("<br>");
}

// ---------- STATS ----------
stats.visits++;
saveAll();
function updateStatsUI(){
const v=document.getElementById("statVisits");
const p=document.getElementById("statPlays");
const c=document.getElementById("statClicks");
if(v) v.textContent=stats.visits;
if(p) p.textContent=stats.plays;
if(c) c.textContent=stats.clicks;
}

// ---------- GLOBAL MESSAGE & MAINTENANCE ----------
const bannerContainer = document.getElementById("bannerContainer");
function renderGlobalMessage(){
const existing = document.getElementById("globalBanner");
if(existing) existing.remove();
if(globalMessage){
const div = document.createElement("div");
div.id="globalBanner";
div.textContent=globalMessage;
bannerContainer.prepend(div);
}
}
function renderMaintenance(){
const existing = document.getElementById("maintBanner");
if(existing) existing.remove();
const statusSpan = document.getElementById("maintenanceStatus");
if(statusSpan) statusSpan.textContent = maintenance ? "ON" : "OFF";
if(maintenance){
const div = document.createElement("div");
div.id="maintBanner";
div.textContent="⚠ Site is in Maintenance Mode";
bannerContainer.prepend(div);
}
}

// ---------- AUTH ----------
const authUser = document.getElementById("authUser");
const authPass = document.getElementById("authPass");
const loginBtn = document.getElementById("loginBtn");
const registerBtn = document.getElementById("registerBtn");
const logoutBtn = document.getElementById("logoutBtn");
const userInfo = document.getElementById("userInfo");

function getRoleLabel(role){
if(role==="admin") return "Admin";
if(role==="mod") return "Moderator";
return "User";
}

function updateUserInfo(){
if(!currentUser){
userInfo.textContent="Not logged in.";
}else{
userInfo.textContent=`Logged in as ${currentUser.username} (${getRoleLabel(currentUser.role)})`;
}
updateAdminVisibility();
}

function findUser(username){
return users.find(u=>u.username.toLowerCase()===username.toLowerCase());
}

function login(username,password){
const u = findUser(username);
if(!u){alert("User not found.");return;}
if(u.banned){alert("You are banned from this site.");return;}
if(u.password!==password){alert("Wrong password.");return;}
currentUser = u;
addLog(`Login: ${u.username} (${u.role})`);
updateUserInfo();
}

function register(username,password){
if(!username || !password){alert("Enter username and password.");return;}
if(findUser(username)){alert("Username already exists.");return;}
const role = "user";
const u = {username,password,role,banned:false};
users.push(u);
saveAll();
addLog(`Registered: ${username}`);
alert("Account created. You can now log in.");
}

loginBtn.addEventListener("click",()=>{
login(authUser.value.trim(),authPass.value);
saveAll();
});
registerBtn.addEventListener("click",()=>{
register(authUser.value.trim(),authPass.value);
});
logoutBtn.addEventListener("click",()=>{
if(currentUser){
addLog(`Logout: ${currentUser.username}`);
}
currentUser=null;
updateUserInfo();
});

// ---------- ADMIN PANEL VISIBILITY ----------
const adminPanel = document.getElementById("adminPanel");
const backBtn = document.getElementById("backBtn");
function canUseAdmin(){
return currentUser && (currentUser.role==="admin" || currentUser.role==="mod");
}
function showAdminPanel(){
if(!canUseAdmin()){alert("Admin/Moderator only.");return;}
adminPanel.style.display="block";
requestAnimationFrame(()=>adminPanel.classList.add("visible"));
}
function hideAdminPanel(){
adminPanel.classList.remove("visible");
setTimeout(()=>{adminPanel.style.display="none";},200);
}
backBtn.addEventListener("click",hideAdminPanel);

function updateAdminVisibility(){
if(!canUseAdmin()){
adminPanel.classList.remove("visible");
adminPanel.style.display="none";
}
}

// ---------- DISGUISED ADMIN TRIGGER ----------
const searchBar = document.getElementById("searchBar");
const SECRET = "admin123";
if(searchBar){
searchBar.addEventListener("keydown",e=>{
if(e.key==="Enter"){
const value = searchBar.value.trim();
if(value===SECRET){
searchBar.value="";
if(canUseAdmin()){
showAdminPanel();
addLog(`Admin panel opened by ${currentUser.username}`);
}else{
alert("You must be admin or moderator to open admin panel.");
}
}else{
addLog(`Search: "${value}" by ${currentUser?currentUser.username:"guest"}`);
alert("Searching for: "+value);
}
}
});
}

// ---------- GAME MANAGER ----------
const gamesGrid = document.getElementById("gamesGrid");
const newGameName = document.getElementById("newGameName");
const addGameBtn = document.getElementById("addGameBtn");
const removeGameSelect = document.getElementById("removeGameSelect");
const removeGameBtn = document.getElementById("removeGameBtn");

function renderGames(){
if(gamesGrid){
gamesGrid.innerHTML="";
games.forEach(name=>{
const card = document.createElement("div");
card.className="game-card";
card.innerHTML=`<div class="game-thumb glow"></div><p>${name}</p>`;
card.addEventListener("click",()=>{
stats.plays++;
saveAll();
updateStatsUI();
addLog(`Game opened: ${name} by ${currentUser?currentUser.username:"guest"}`);
if(name==="Neon Clicker") showSection("game1");
else alert("Placeholder game: "+name);
});
gamesGrid.appendChild(card);
});
}
if(removeGameSelect){
removeGameSelect.innerHTML="";
games.forEach(name=>{
const opt=document.createElement("option");
opt.value=name;opt.textContent=name;
removeGameSelect.appendChild(opt);
});
}
}

addGameBtn.addEventListener("click",()=>{
if(!canUseAdmin()){alert("Admin/Moderator only.");return;}
const name = newGameName.value.trim();
if(!name) return;
if(games.includes(name)){alert("Game already exists.");return;}
games.push(name);
saveAll();
renderGames();
addLog(`Game added: ${name} by ${currentUser.username}`);
newGameName.value="";
});
removeGameBtn.addEventListener("click",()=>{
if(!canUseAdmin()){alert("Admin/Moderator only.");return;}
const name = removeGameSelect.value;
if(name==="Neon Clicker"){alert("Cannot remove core game.");return;}
games = games.filter(g=>g!==name);
saveAll();
renderGames();
addLog(`Game removed: ${name} by ${currentUser.username}`);
});

// ---------- USERS & ROLES / KICK / BAN ----------
const userSelect = document.getElementById("userSelect");
const promoteModBtn = document.getElementById("promoteModBtn");
const demoteUserBtn = document.getElementById("demoteUserBtn");
const kickBtn = document.getElementById("kickBtn");
const banBtn = document.getElementById("banBtn");
const unbanBtn = document.getElementById("unbanBtn");

function renderUserList(){
if(!userSelect) return;
userSelect.innerHTML="";
users.forEach(u=>{
const opt=document.createElement("option");
opt.value=u.username;
opt.textContent=`${u.username} (${getRoleLabel(u.role)}${u.banned?" - BANNED":""})`;
userSelect.appendChild(opt);
});
}

function requireAdmin(){
if(!currentUser || currentUser.role!=="admin"){
alert("Admin only.");
return false;
}
return true;
}

promoteModBtn.addEventListener("click",()=>{
if(!requireAdmin())return;
const uname=userSelect.value;
const u=findUser(uname);
if(!u)return;
if(u.username===ADMIN_USERNAME){alert("Admin is already highest role.");return;}
u.role="mod";
saveAll();
renderUserList();
addLog(`Promoted to mod: ${u.username} by ${currentUser.username}`);
});

demoteUserBtn.addEventListener("click",()=>{
if(!requireAdmin())return;
const uname=userSelect.value;
const u=findUser(uname);
if(!u)return;
if(u.username===ADMIN_USERNAME){alert("Cannot demote admin.");return;}
u.role="user";
saveAll();
renderUserList();
addLog(`Demoted to user: ${u.username} by ${currentUser.username}`);
});

kickBtn.addEventListener("click",()=>{
if(!canUseAdmin()){alert("Admin/Moderator only.");return;}
const uname=userSelect.value;
if(!uname)return;
addLog(`Kick: ${uname} by ${currentUser.username}`);
if(currentUser && currentUser.username===uname){
currentUser=null;
updateUserInfo();
}
alert(`User ${uname} has been kicked (local session only).`);
});

banBtn.addEventListener("click",()=>{
if(!canUseAdmin()){alert("Admin/Moderator only.");return;}
const uname=userSelect.value;
const u=findUser(uname);
if(!u)return;
if(u.username===ADMIN_USERNAME){alert("Cannot ban admin.");return;}
u.banned=true;
saveAll();
renderUserList();
addLog(`Banned: ${u.username} by ${currentUser.username}`);
if(currentUser && currentUser.username===u.username){
currentUser=null;
updateUserInfo();
alert("You have been banned.");
}
});

unbanBtn.addEventListener("click",()=>{
if(!canUseAdmin()){alert("Admin/Moderator only.");return;}
const uname=userSelect.value;
const u=findUser(uname);
if(!u)return;
u.banned=false;
saveAll();
renderUserList();
addLog(`Unbanned: ${u.username} by ${currentUser.username}`);
});

// ---------- GLOBAL MESSAGE CONTROLS ----------
const globalMsgInput = document.getElementById("globalMsgInput");
const setGlobalMsgBtn = document.getElementById("setGlobalMsgBtn");
const clearGlobalMsgBtn = document.getElementById("clearGlobalMsgBtn");

setGlobalMsgBtn.addEventListener("click",()=>{
if(!canUseAdmin()){alert("Admin/Moderator only.");return;}
globalMessage = globalMsgInput.value.trim();
saveAll();
renderGlobalMessage();
addLog(`Global message set by ${currentUser.username}`);
});
clearGlobalMsgBtn.addEventListener("click",()=>{
if(!canUseAdmin()){alert("Admin/Moderator only.");return;}
globalMessage="";
saveAll();
renderGlobalMessage();
addLog(`Global message cleared by ${currentUser.username}`);
});

// ---------- MAINTENANCE ----------
const toggleMaintenanceBtn = document.getElementById("toggleMaintenance");
toggleMaintenanceBtn.addEventListener("click",()=>{
if(!canUseAdmin()){alert("Admin/Moderator only.");return;}
maintenance=!maintenance;
saveAll();
renderMaintenance();
addLog(`Maintenance mode: ${maintenance?"ON":"OFF"} by ${currentUser.username}`);
});

// ---------- NEON CLICKER ----------
const scoreEl = document.getElementById("score");
const clickBtn = document.getElementById("clickBtn");
let score=0;
if(clickBtn && scoreEl){
clickBtn.addEventListener("click",()=>{
score++;
scoreEl.textContent=score;
stats.clicks++;
saveAll();
updateStatsUI();
});
}

// ---------- INIT ----------
renderLogs();
renderGames();
renderUserList();
updateStatsUI();

renderMaintenance();
updateUserInfo();
